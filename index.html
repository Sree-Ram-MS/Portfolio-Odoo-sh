# -*- coding: utf-8 -*-
from odoo import models, fields, api, _
from odoo.exceptions import UserError

class CustomSalesOrder(models.Model):
    """ Inherits sale.order to add custom APIs and AI logs """
    _inherit = 'sale.order'

    payment_status = fields.Selection([
        ('pending', 'Pending'),
        ('paid', 'Paid'),
        ('failed', 'Failed')
    ], string="Payment Status", default='pending', tracking=True)

    @api.model
    def create(self, vals):
        # Custom logic for API triggers during sale creation
        res = super(CustomSalesOrder, self).create(vals)
        if res.amount_total > 5000:
            res.message_post(body=_("High-value sales order created. Processing payment sync..."))
        return res

    def action_sync_payment_gateway(self):
        """ Core Integration Method: Synchronizes payment API """
        self.ensure_one()
        try:
            payload = {
                'order_id': self.name,
                'amount': self.amount_total,
                'currency': self.currency_id.name
            }
            # Call custom integrated payment gateway client
            response = self.env['payment.gateway.client'].post('/v1/charges', payload)
            if response.get('status') == 'success':
                self.write({'payment_status': 'paid'})
            else:
                self.write({'payment_status': 'failed'})
        except Exception as e:
            raise UserError(_("Payment sync failed: %s") % str(e))
